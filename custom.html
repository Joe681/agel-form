# 如何二次封装组件

TIP

agel-form 组件可以灵活搭配第三方或者自定义的组件进行使用，唯一需要注意的是避免组件的动态属性和现有的 Item 属性产生命名冲突，除此之外没有其他额外的规则，下面是一个自定义二次封装组件的简单教程。

例子以简单二次封装 el-select 组件为例子：

  1. 第一步：创建 .vue 文件，设置一个基础模板，定义好 props，建议 inheritAttrs 设置为 false。
<template>
</template>

<script>
export default {
  name: "custom-select",
  inheritAttrs: false,
  props: {
    value: [String, Number, Array],
    options: Array,
  },
  install(vue) {
    vue.component(this.name, this);
  },
};
</script>
  1. 第二步：绑定 value，挂载动态属性 $attrs (opens new window), 绑定动态事件 $listeners (opens new window)，写自定义模板逻辑。

 
 
 
 
















<template>
  <el-select :value="value" v-bind="$attrs" v-on="$listeners">
    <el-option v-for="option of options" v-bind='option' :key="option.label" :label="option.label" :value="option.value">
    </el-option>
  </el-select>
</template>

<script>
export default {
  name: "custom-select",
  inheritAttrs: false,
  props: {
    value: [String, Number, Array],
    options: Array,
  },
  install(vue) {
    vue.component(this.name, this);
  },
};
</script>
  1. 第三步：根据需要，添加插槽，一个简单的二次封装组件就大功告成了！




 
 
 

















<template>
  <el-select :value="value" v-bind="$attrs" v-on="$listeners">
    <el-option v-for="option of options" v-bind='option' :key="option.label" :label="option.label" :value="option.value">
    </el-option>
    <template v-slot:prefix>
      <slot name="prefix"></slot>
    </template>
  </el-select>
</template>

<script>
export default {
  name: "custom-select",
  inheritAttrs: false,
  props: {
    value: [String, Number, Array],
    options: Array,
  },
  install(vue) {
    vue.component(this.name, this);
  },
};
</script>
  1. 第四步：注册到全局，配合 agel-form 使用。
import Vue from 'vue';
import customSelect from "@/components/customSelect"

Vue.use(customSelect);

<template>
  <div class="demo border">
    <agel-form v-model="form"></agel-form>
  </div>
</template>
 
<script>
export default {
  data() {
    return {
      form: {
        data: {},
        items: {
          region: {
            component: "custom-select",
            label: "活动区域",
            options: [
              { label: "区域1", value: "区域1" },
              { label: "区域2", value: "区域2" },
              { label: "区域3", value: "区域3" },
            ],
          },
        },
      },
    };
  },
};
</script>
